<script>

  const slides = [

    { title: "I",    id: "1172849073" },

    { title: "II",   id: "1172849046" },

    { title: "III",  id: "1172849012" },

    { title: "IV",   id: "1172848986" },

    { title: "V",    id: "1172848954" },

    { title: "VI",   id: "1172848930" },

    { title: "VII",  id: "1172848889" },

    { title: "VIII", id: "1172848857" },

    { title: "IX",   id: "1172848833" },

    { title: "X",    id: "1172848807" },

    { title: "XI",   id: "1172848786" },

    { title: "XII",  id: "1172848769" },

    { title: "XIII", id: "1172848710" },

    { title: "XIV",  id: "1172848667" },

    { title: "XV",   id: "1172848640" },

  ];


  let current = 0, transitioning = false, soundOn = false;

  let idleTimer, ghostTimer;

  let players = [];


  const reel = document.getElementById('reel');

  const dotsEl = document.getElementById('dots');


  // Build slides

  const slideEls = slides.map((s, i) => {

    const el = document.createElement('div');

    el.className = 'slide' + (i === 0 ? ' active' : '');

    // Vimeo embed: autoplay, loop, muted, no controls, no title, no byline

    const src = `https://player.vimeo.com/video/${s.id}?autoplay=1&loop=1&muted=1&background=1&autopause=0&transparent=0`;

    el.innerHTML = `

      <div class="vimeo-wrap">

        <iframe src="${src}" allow="autoplay; fullscreen" allowfullscreen></iframe>

      </div>

      <div class="project-title">${s.title}</div>`;

    reel.appendChild(el);

    return el;

  });


  // Build dots

  slides.forEach((_, i) => {

    const d = document.createElement('div');

    d.className = 'dot' + (i === 0 ? ' active' : '');

    dotsEl.appendChild(d);

  });


  function updateUI() {

    document.getElementById('counter-current').textContent = String(current+1).padStart(2,'0');

    document.getElementById('counter-progress').style.width = ((current+1)/slides.length*100)+'%';

    [...dotsEl.querySelectorAll('.dot')].forEach((d,i) => d.className = 'dot'+(i===current?' active':''));

  }


  const overlay = document.getElementById('overlay');

  function goTo(idx) {

    if (transitioning || idx === current) return;

    transitioning = true;

    overlay.classList.add('flash');

    setTimeout(() => {

      slideEls[current].classList.remove('active');

      current = idx;

      slideEls[current].classList.add('active');

      updateUI();

      overlay.classList.remove('flash');

      setTimeout(() => { transitioning = false; }, 1200);

    }, 380);

  }

  const next = () => goTo((current+1) % slides.length);

  const prev = () => goTo((current-1+slides.length) % slides.length);


  // Scroll

  let acc = 0;

  window.addEventListener('wheel', e => {

    e.preventDefault();

    acc += e.deltaY;

    if (Math.abs(acc) > 55) { acc > 0 ? next() : prev(); acc = 0; }

  }, { passive: false });


  // Touch

  let ty = 0;

  window.addEventListener('touchstart', e => ty = e.touches[0].clientY);

  window.addEventListener('touchend', e => {

    const dy = ty - e.changedTouches[0].clientY;

    if (Math.abs(dy) > 40) dy > 0 ? next() : prev();

  });


  // Keyboard

  window.addEventListener('keydown', e => {

    if (e.key === 'ArrowDown' || e.key === 'ArrowRight') next();

    if (e.key === 'ArrowUp' || e.key === 'ArrowLeft') prev();

  });


  // Identity idle

  const identity = document.getElementById('identity');

  function resetIdle() {

    clearTimeout(idleTimer); clearTimeout(ghostTimer);

    identity.classList.remove('fade','ghost');

    idleTimer = setTimeout(() => {

      identity.classList.add('fade');

      ghostTimer = setTimeout(() => {

        identity.classList.remove('fade');

        identity.classList.add('ghost');

      }, 1600);

    }, 3000);

  }

  window.addEventListener('mousemove', resetIdle);

  window.addEventListener('wheel', resetIdle);

  resetIdle();


  // Hide dots on first scroll

  window.addEventListener('wheel', () => dotsEl.classList.add('hidden'), { once: true });


  // Sound — uses Vimeo postMessage API

  function toggleSound() {

    soundOn = !soundOn;

    document.getElementById('sound').classList.toggle('on', soundOn);

    document.getElementById('sound-label').textContent = soundOn ? 'sound on' : 'sound off';

    document.querySelectorAll('.vimeo-wrap iframe').forEach(iframe => {

      const msg = soundOn

        ? JSON.stringify({ method: 'setVolume', value: 1 })

        : JSON.stringify({ method: 'setVolume', value: 0 });

      iframe.contentWindow.postMessage(msg, '*');

    });

  }


  // Cursor

  const cur = document.getElementById('cursor');

  document.addEventListener('mousemove', e => {

    cur.style.left = e.clientX+'px'; cur.style.top = e.clientY+'px';

    cur.classList.remove('hidden');

  });

  document.addEventListener('mouseleave', () => cur.classList.add('hidden'));

  document.getElementById('sound').addEventListener('mouseenter', () => cur.classList.add('hover'));

  document.getElementById('sound').addEventListener('mouseleave', () => cur.classList.remove('hover'));


  updateUI();

</script>
